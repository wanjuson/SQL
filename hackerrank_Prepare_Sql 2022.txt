-- The PADS
SELECT
    CONCAT(A.NAME, "(", LEFT(A.OCCUPATION,1), ")") AS AA
FROM OCCUPATIONS A
ORDER BY A.NAME
;
SELECT
    CONCAT("There are a total of ", B.OCC_CNT, " ", LOWER(B.OCCUPATION), "s.") AS AA
FROM 
    (
        SELECT OCCUPATION, COUNT(NAME) AS OCC_CNT
        FROM OCCUPATIONS
        GROUP BY OCCUPATION
    ) B
ORDER BY B.OCC_CNT ASC, B.OCCUPATION 
;
