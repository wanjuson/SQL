-- 조건에 맞는 사용자와 총 거래금액 조회하기
SELECT
    USED_GOODS_USER.USER_ID
    , USED_GOODS_USER.NICKNAME
    , SUM(USED_GOODS_BOARD.PRICE) TOTAL_SALES
FROM USED_GOODS_BOARD 
    LEFT JOIN USED_GOODS_USER 
        ON USED_GOODS_BOARD.WRITER_ID = USED_GOODS_USER.USER_ID

WHERE USED_GOODS_BOARD.STATUS = 'DONE'
GROUP BY 1,2
HAVING SUM(USED_GOODS_BOARD.PRICE) >= 700000
ORDER BY TOTAL_SALES
;
