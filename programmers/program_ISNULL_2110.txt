program_ISNULL_2110

-- 이름이 없는 동물의 아이디
SELECT ANIMAL_ID
FROM ANIMAL_INS 
WHERE 1=1
    AND NAME IS NULL
ORDER BY ANIMAL_ID ASC
;

-- 이름이 있는 동물의 아이디
SELECT ANIMAL_ID
FROM ANIMAL_INS 
WHERE 1=1
    AND NAME IS NOT NULL
ORDER BY ANIMAL_ID ASC
;

-- NULL 처리하기 check
SELECT 
    ANIMAL_TYPE
    , CASE WHEN NAME IS NULL THEN "No name"
        ELSE NAME END AS NAME
    , SEX_UPON_INTAKE
FROM ANIMAL_INS 
ORDER BY ANIMAL_ID
;
