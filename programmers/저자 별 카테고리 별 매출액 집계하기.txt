-- 저자 별 카테고리 별 매출액 집계하기

SELECT
    BOOK.AUTHOR_ID
    , AUTHOR.AUTHOR_NAME
    , BOOK.CATEGORY
    , SUM(BOOK.PRICE * BOOK_SALES.SALES) TOTAL_SALES
FROM BOOK
    LEFT JOIN AUTHOR 
        ON BOOK.AUTHOR_ID = AUTHOR.AUTHOR_ID
    LEFT JOIN BOOK_SALES  
        ON BOOK.BOOK_ID = BOOK_SALES.BOOK_ID
        AND BOOK_SALES.SALES_DATE BETWEEN DATE('2022-01-01') AND DATE('2022-01-31')
GROUP BY 1,2,3
ORDER BY BOOK.AUTHOR_ID ASC, BOOK.CATEGORY DESC
;
